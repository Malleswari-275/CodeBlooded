1class Solution {
2public:
3    int minOperations(vector<int>& nums) {
4        unordered_map<int,int>mp;
5        int ans = 0;
6        for(int i = nums.size()-1; i >= 0; i--){
7            if(mp.find(nums[i]) != mp.end()){
8                ans = (i/3) +1;
9                break;
10            }
11            mp[nums[i]]++;
12        }
13        return ans;
14    }
