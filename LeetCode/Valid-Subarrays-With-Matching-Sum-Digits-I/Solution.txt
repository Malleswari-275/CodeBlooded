1class Solution {
2public:
3    long long Digit(long long n){
4        while(n >= 10){
5            n/=10;
6        }
7        return n;
8    }
9    int countValidSubarrays(vector<int>& nums, int x) {
10        long long cnt = 0;
11        vector<long long>pref(nums.size());
12        pref[0] = nums[0];
13        for(int i = 1; i < nums.size(); i++) pref[i] = pref[i-1]+nums[i];
14        for(int i =0; i < nums.size(); i++){
15            for(int j = i; j < nums.size(); j++){
16                long long n = (i == 0) ? pref[j] : pref[j]-pref[i-1];
17                long long last = n%10;
18                long long first = Digit(n);
19                if(first == x && last == x)cnt++;
20            }
21        }
22        return cnt;
23    }
24};
